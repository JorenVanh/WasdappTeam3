CREATE schema wasdapp;

titel varchar(40)  not null primary key,
locatie varchar(50),
straat varchar(50), 
nummer varchar(45),
postcode varchar(45),
gemeente varchar(45),
land varchar(40),
omschrijving varchar(120),
wikipedia_link varchar(35),
website varchar(40),
telefoon varchar(25),
e_mail varchar(40),
prijs double,
persoon varchar(50)
);
